   43 data FrontendResult v x = FrontendResult
   44     { frDataFlow :: DataFlowGraph v x
   45     , frTrace :: [TraceVar]

   49 data TraceVar = TraceVar {tvFmt, tvVar :: Text}
   50     deriving (Show)

   52 defaultFmt = "%.3f"

   54 -- |List contains IO Function names to be printed by default.
   55 listFuncIO = ["send", "recieve"]

   57 prettyLog traceVars hms = map prettyHM hms
   58     where
   59         prettyHM hm = HM.fromList $ map (fromMaybe undefined) $ filter isJust $ map prettyX $ HM.toList hm
   60         prettyX (v0, x) = do
   61             -- variables names end on #0, #1..., so we trim this suffix
   62             let v = takeWhile (/= '#') $ toString v0
   63             fmt <- v2fmt M.!? v
   64             Just (toString v, printx (T.unpack fmt) x)
   65         v2fmt = M.fromList $ map (\(TraceVar fmt v) -> (toString v, fmt)) traceVars
   66         printx p x
   67             | 'f' `elem` p = printf p (fromRational (toRational x) :: Double)
   68             | 's' `elem` p = printf p $ show x
   69             | otherwise = printf p x

   71 lua2functions src =
   72     let ast = either (\e -> error $ "can't parse lua src: " ++ show e) id $ parseText chunk src
   73         AlgBuilder{algItems} = buildAlg ast
   74         fs = filter (\case Function{} -> True; _ -> False) algItems
   75         ioVariables = getIOVariables defaultFmt algItems
   76         varDict =
   77             M.fromList $
   78                 map varRow $
   79                     group $ sort $ concatMap fIn fs
   80         alg = snd $ execState (mapM_ (store <=< function2nitta) fs) (varDict, [])
   81         frDataFlow = fsToDataFlowGraph alg
   82         traceFunctions = [tf | tf@TraceFunction{} <- algItems]
   83 
   84         frTrace =
   85             if not $ null traceFunctions
   86                 then
   87                     [ TraceVar tFmt v
   88                     | TraceFunction{tFmt, tVars} <- traceFunctions
   89                     , v <- tVars
   90                     ]
   91                 else
   92                     ioVariables
   93                         <> [ TraceVar defaultFmt iVar
   94                            | InputVar{iVar} <- algItems
   95                            ]
   96      in FrontendResult
   97             { frDataFlow
   98             , frTrace
   99             , frPrettyLog = prettyLog frTrace
  100             }
  101     where
  102         varRow lst@(x : _) =
  103             let vs = zipWith f lst [0 :: Int ..]
  104              in (x, (vs, vs))
  105         varRow _ = undefined
  106         f v ix = fromString (toString v <> "#" <> show ix)
  107 
  108 getIOVariables fmt algItems = concatMap convToTraceVar filterIoFunc
  109     where
  110         convToTraceVar func = map (TraceVar fmt) $ fIn func
  111         filterIoFunc = filter (\case item@Function{} -> fName item `elem` listFuncIO; _ -> False) algItems
  112 
  113 buildAlg ast = flip execState st0 $ do
  114     addMainInputs mainFunDef mainCall
  115     mapM_ (processStatement mainName) $ funAssignStatements mainFunDef
  116     addConstants
  117     where
  118         Right (mainName, mainCall, mainFunDef) = findMain ast
  119         st0 =
  120             AlgBuilder
  121                 { algItems = []
  122                 , algBuffer = []
  123                 , algVarGen = map (pack . show) [0 ..]
  124                 , algVars = []
  125                 }

  127 data AlgBuilder x = AlgBuilder
  128     { algItems :: [AlgBuilderItem x]
  129     , algBuffer :: [AlgBuilderItem x]
  130     , algVarGen :: [Text]
  131     , algVars :: [Text]

  134 instance (Show x) => Show (AlgBuilder x) where
  135     show AlgBuilder{algItems, algBuffer, algVars} =
  136         "AlgBuilder\n{ algItems=\n"
  137             ++ S.join "\n" (map (("    " <>) . show) $ reverse algItems)
  138             ++ "\nalgBuffer: "
  139             ++ show algBuffer
  140             ++ "\nalgVars: "
  141             ++ show algVars
  142             ++ "\n}"

  144 data AlgBuilderItem x
  145     = InputVar
  146         { -- |Index
  147           iIx :: Int
  148         , -- |Symbol

  150         , -- |Initial value
  151           iX :: x

  153     | Constant {cVar :: Text, cX :: x, cTextX :: Text}
  154     | Alias {aFrom :: Text, aTo :: Text}
  155     | Renamed {rFrom :: Text, rTo :: Text}
  156     | Function
  157         { fIn :: [Text]
  158         , fOut :: [Text]
  159         , fName :: String
  160         , fValues :: [x]
  161         , fInt :: [Int]

  163     | TraceFunction
  164         { tVars :: [Text]
  165         , tFmt :: Text
  166         }
  167     -- FIXME: check all local variable declaration

  170 -- *Translate AlgBuiler functions to nitta functions
  171 function2nitta Function{fName = "loop", fIn = [i], fOut = [o], fValues = [x], fInt = []} = F.loop x <$> input i <*> output o
  172 function2nitta Function{fName = "buffer", fIn = [i], fOut = [o], fValues = [], fInt = []} = F.buffer <$> input i <*> output o
  173 function2nitta Function{fName = "brokenBuffer", fIn = [i], fOut = [o], fValues = [], fInt = []} = F.brokenBuffer <$> input i <*> output o
  174 function2nitta Function{fName = "constant", fIn = [], fOut = [o], fValues = [x], fInt = []} = F.constant x <$> output o
  175 function2nitta Function{fName = "send", fIn = [i], fOut = [], fValues = [], fInt = []} = F.send <$> input i
  176 function2nitta Function{fName = "add", fIn = [a, b], fOut = [c], fValues = [], fInt = []} = F.add <$> input a <*> input b <*> output c
  177 function2nitta Function{fName = "sub", fIn = [a, b], fOut = [c], fValues = [], fInt = []} = F.sub <$> input a <*> input b <*> output c
  178 function2nitta Function{fName = "multiply", fIn = [a, b], fOut = [c], fValues = [], fInt = []} = F.multiply <$> input a <*> input b <*> output c
  179 function2nitta Function{fName = "divide", fIn = [d, n], fOut = [q, r], fValues = [], fInt = []} = F.division <$> input d <*> input n <*> output q <*> output r
  180 function2nitta Function{fName = "receive", fIn = [], fOut = [o], fValues = [], fInt = []} = F.receive <$> output o
  181 function2nitta Function{fName = "shiftL", fIn = [a], fOut = [c], fValues = [], fInt = [s]} = F.shiftL s <$> input a <*> output c
  182 function2nitta Function{fName = "shiftR", fIn = [a], fOut = [c], fValues = [], fInt = [s]} = F.shiftR s <$> input a <*> output c
  183 function2nitta f = error $ "frontend don't known function: " ++ show f

  185 input v = do
  186     (dict, fs) <- get
  187     let (x : xs, lst) = dict M.! v
  188     put (M.insert v (xs, lst) dict, fs)
  189     return x

  191 output v
  192     | T.head v == '_' = return []
  193     | otherwise = gets $ \(dict, _fs) ->
  194         snd (fromMaybe (error $ "unknown variable: " <> show v <> " defined: " <> show (M.keys dict)) (dict M.!? v))

  196 store f = modify'_ $ second (f :)

  199 findMain (Block statements Nothing)
  200     | [call] <- filter (\case FunCall{} -> True; _ -> False) statements
  201       , [funAssign] <- filter (\case FunAssign{} -> True; _ -> False) statements
  202       , (FunCall (NormalFunCall (PEVar (VarName (Name fnCall))) _)) <- call
  203       , (FunAssign (FunName (Name fnAssign) _ _) _) <- funAssign
  204       , fnCall == fnAssign =
  205         Right (fnCall, call, funAssign)
  206 findMain _ = error "can't find main function in lua source code"
  207 
  208 addMainInputs
  209     (FunAssign (FunName (Name _funName) _ _) (FunBody declArgs _ _))
  210     (FunCall (NormalFunCall _ (Args callArgs))) = do
  211         let vars = map (\case (Name v) -> v) declArgs
  212 
  213         let values = map (\case (Number _ s) -> unpack s; _ -> error "lua: wrong main argument") callArgs
  214         let values' = map read values
  215         when (length vars /= length values') $
  216             error "a different number of arguments in main a function declaration and call"
  217 
  218         forM_ (zip3 [0 ..] vars values') $
  219             \(iIx, iVar, iX) -> addItem InputVar{iIx, iVar, iX} [iVar]
  220 addMainInputs _ _ = error "bad main function description"
  221 
  222 addConstants = do
  223     AlgBuilder{algItems} <- get
  224     let constants = filter (\case Constant{} -> True; _ -> False) algItems
  225     forM_ constants $ \Constant{cVar, cX} ->
  226         addFunction Function{fName = "constant", fIn = [], fOut = [cVar], fValues = [cX], fInt = []}
  227 
  228 parseLeftExp = map $ \case
  229     VarName (Name v) -> v
  230     e -> error $ "bad left expression: " ++ show e
  231 
  232 -- define a local variable: @local x@ or @local x = ...@. We ignore it, because
  233 -- don't need to support scopes.
  234 processStatement _fn (LocalAssign _names Nothing) =
  235     return ()
  236 processStatement fn (LocalAssign names (Just rexp)) =
  237     processStatement fn $ Assign (map VarName names) rexp
  238 processStatement fn (Assign lexps@[_] [Unop Neg (Number ntype ntext)]) =
  239     processStatement fn (Assign lexps [Number ntype ("-" <> ntext)])
  240 processStatement _fn (Assign lexps@[_] [n@(Number _ textX)]) = do
  241     let outs@[name] = parseLeftExp lexps
  242     diff <- renameVarsIfNeeded outs
  243     AlgBuilder{algItems} <- get
  244     case findConstant textX algItems of
  245         Just Constant{cVar} ->
  246             -- TODO: sould work correctly, see test test_lua_two_name_for_same_constant
  247             error $
  248                 concat
  249                     [ "using several names for one constant value ("
  250                     , show cVar
  251                     , " and "
  252                     , show name
  253                     , "), please, use only one name (temporal restriction)"
  254                     ]
  255         Nothing -> do
  256             -- `expConstant` return a wrong name, the correct name will be established
  257             -- after all statement was processed
  258             void $ expConstant name n
  259             flushBuffer diff []
  260         _ -> error "internal error"
  261 
  262 -- e.g. @n, d = a / b@, or @n, d = f()@
  263 processStatement _fn (Assign lexps [rexp])
  264     | length lexps > 1 = do
  265         let outs = parseLeftExp lexps
  266         diff <- renameVarsIfNeeded outs
  267         rightExp diff outs rexp
  268         flushBuffer diff outs
  269 
  270 -- e.g. @a = 1@ or @a, b = 1, 2@
  271 processStatement _fn st@(Assign lexps rexps)
  272     | length lexps == length rexps = do
  273         let outs = parseLeftExp lexps
  274         diff <- renameVarsIfNeeded outs
  275         zipWithM_ (rightExp diff) (map (: []) outs) rexps
  276         flushBuffer diff outs
  277     | otherwise = error $ "assignment mismatch: " ++ show st
  278 -- recursive call of main function
  279 processStatement fn (FunCall (NormalFunCall (PEVar (VarName (Name fName))) (Args args)))
  280     | fn == fName =
  281         do
  282             AlgBuilder{algItems} <- get
  283             let algIn = reverse $ filter (\case InputVar{} -> True; _ -> False) algItems
  284             mapM_ (uncurry f) $ zip algIn args
  285     where
  286         f InputVar{iX, iVar} rexp = do
  287             i <- expArg [] rexp
  288             let loop = Function{fName = "loop", fIn = [i], fOut = [iVar], fValues = [iX], fInt = []}
  289             modify'_ $ \alg@AlgBuilder{algItems} -> alg{algItems = loop : algItems}
  290         f _ _ = undefined
  291 processStatement _fn (FunCall (NormalFunCall (PEVar (SelectName (PEVar (VarName (Name "debug"))) (Name fName))) (Args args))) = do
  292     fIn <- mapM (expArg []) args
  293     case (fName, fIn) of
  294         ("trace", tFmt : vs)
  295             | T.isPrefixOf "\"" tFmt && T.isPrefixOf "\"" tFmt ->
  296                 addFunction TraceFunction{tVars = vs, tFmt = T.replace "\"" "" tFmt}
  297         ("trace", vs) -> addFunction TraceFunction{tVars = vs, tFmt = defaultFmt}
  298         _ -> error $ "unknown debug method: " ++ show fName ++ " " ++ show args
  299 processStatement _fn (FunCall (NormalFunCall (PEVar (VarName (Name fName))) (Args args))) = do
  300     fIn <- mapM (expArg []) args
  301     addFunction Function{fName = unpack fName, fIn, fOut = [], fValues = [], fInt = []}
  302 processStatement _fn st = error $ "statement: " ++ show st
  303 
  304 rightExp diff fOut (Binop ShiftL a (Number IntNum s)) = do
  305     a' <- expArg diff a
  306     let f = Function{fName = "shiftL", fIn = [a'], fOut, fValues = [], fInt = [read $ unpack s :: Int]}
  307     patchAndAddFunction f diff
  308 rightExp diff fOut (Binop ShiftR a (Number IntNum s)) = do
  309     a' <- expArg diff a
  310     let f = Function{fName = "shiftR", fIn = [a'], fOut, fValues = [], fInt = [read $ unpack s :: Int]}
  311     patchAndAddFunction f diff
  312 rightExp diff fOut (Binop op a b) = do
  313     a' <- expArg diff a
  314     b' <- expArg diff b
  315     let f = Function{fName = binop op, fIn = [a', b'], fOut, fValues = [], fInt = []}
  316     patchAndAddFunction f diff
  317     where
  318         binop Add = "add"
  319         binop Sub = "sub"
  320         binop Mul = "multiply"
  321         binop Div = "divide"
  322         binop o = error $ "unknown binop: " ++ show o
  323 rightExp
  324     diff
  325     fOut
  326     ( PrefixExp
  327             ( PEFunCall
  328                     ( NormalFunCall
  329                             (PEVar (VarName (Name fn)))
  330                             (Args args)
  331                         )
  332                 )
  333         ) = do
  334         fIn <- mapM (expArg diff) args
  335         let f = Function{fName = unpack fn, fIn, fOut, fValues = [], fInt = []}
  336         patchAndAddFunction f diff
  337 rightExp diff [a] (PrefixExp (PEVar (VarName (Name b)))) -- a = b
  338     =
  339     addItemToBuffer Alias{aFrom = a, aTo = applyPatch diff b}
  340 -- a = (...)
  341 rightExp diff out (PrefixExp (Paren e)) = rightExp diff out e
  342 rightExp diff [a] n@(Number _ _) = rightExp diff [a] (PrefixExp (PEFunCall (NormalFunCall (PEVar (VarName (Name "buffer"))) (Args [n]))))
  343 rightExp diff [a] (Unop Neg (Number numType n)) = rightExp diff [a] (PrefixExp (PEFunCall (NormalFunCall (PEVar (VarName (Name "buffer"))) (Args [Number numType $ T.cons '-' n]))))
  344 rightExp diff [a] (Unop Neg expr@(PrefixExp _)) =
  345     -- FIXME: add negative function
  346     let binop = Binop Sub (Number IntNum "0") expr
  347      in rightExp diff [a] binop
  348 rightExp _diff _out rexp = error $ "rightExp: " ++ show rexp
  349 
  350 expArg _diff n@(Number _ n') = expConstant (n' <> "@const") n
  351 expArg _diff (Unop Neg (Number numType n)) =
  352     let n' = "-" <> n in expConstant (n' <> "@const") $ Number numType n'
  353 expArg _diff (String s) = return s
  354 expArg _diff (PrefixExp (PEVar (VarName (Name var)))) = findAlias var
  355 expArg diff call@(PrefixExp (PEFunCall _)) = do
  356     c <- genVar "tmp"
  357     rightExp diff [c] call
  358     return c
  359 expArg diff (PrefixExp (Paren arg)) = expArg diff arg
  360 expArg diff binop@Binop{} = do
  361     c <- genVar "tmp"
  362     rightExp diff [c] binop
  363     return c
  364 expArg diff (Unop Neg expr@(PrefixExp _)) = do
  365     c <- genVar "tmp"
  366     let binop = Binop Sub (Number IntNum "0") expr
  367     rightExp diff [c] binop
  368     return c
  369 expArg _diff a = error $ "expArg: " ++ show a

  372 findConstant textX = find (\case Constant{cTextX} | cTextX == textX -> True; _ -> False)
  373 
  374 expConstant name (Number _ textX) = do
  375     AlgBuilder{algItems} <- get
  376     case findConstant textX algItems of
  377         Just Constant{cVar} -> return cVar
  378         Nothing -> do
  379             addItem
  380                 Constant
  381                     { cX = read $ unpack textX
  382                     , cVar = name
  383                     , cTextX = textX
  384                     }
  385                 []
  386             return name
  387         Just _ -> error "internal error"
  388 expConstant _ _ = undefined
  389 
  390 addFunction f@Function{fOut} = do
  391     diff <- renameVarsIfNeeded fOut
  392     patchAndAddFunction f diff
  393 addFunction f@TraceFunction{} =
  394     modify'_ $ \alg@AlgBuilder{algItems} ->
  395         alg{algItems = f : algItems}
  396 addFunction _ = error "addFunction error"
  397 
  398 patchAndAddFunction f@Function{fIn} diff = do
  399     let fIn' = map (applyPatch diff) fIn
  400     modify'_ $ \alg@AlgBuilder{algItems} ->
  401         alg{algItems = f{fIn = fIn'} : algItems}
  402 patchAndAddFunction _ _ = undefined
  403 
  404 renameVarsIfNeeded fOut = do
  405     AlgBuilder{algVars} <- get
  406     mapM autoRename $ filter (`elem` algVars) fOut
  407 
  408 autoRename var = do
  409     var' <- genVar var
  410     renameFromTo var var'
  411     return (var, var')
  412 
  413 renameFromTo rFrom rTo = do
  414     alg@AlgBuilder{algItems, algVars} <- get
  415     put
  416         alg
  417             { algItems = Renamed{rFrom, rTo} : patch algItems
  418             , algVars = rTo : algVars
  419             }
  420     where
  421         patch [] = []
  422         patch (i@InputVar{iVar} : xs) = i{iVar = rn iVar} : patch xs
  423         patch (Constant{cX, cVar, cTextX} : xs) = Constant{cX, cVar = rn cVar, cTextX} : patch xs
  424         patch (Alias{aFrom, aTo} : xs) = Alias (rn aFrom) (rn aTo) : patch xs
  425         patch (f@Function{fIn, fOut} : xs) = f{fIn = map rn fIn, fOut = map rn fOut} : patch xs
  426         patch (x : xs) = x : patch xs
  427 
  428         rn v
  429             | v == rFrom = rTo
  430             | otherwise = v
  431 
  432 funAssignStatements (FunAssign _ (FunBody _ _ (Block statments _))) = statments
  433 funAssignStatements _ = error "funAssignStatements : not function assignment"
  434 
  435 flushBuffer diff outs = modify'_ $
  436     \alg@AlgBuilder{algBuffer, algItems, algVars} ->
  437         alg
  438             { algItems = algBuffer ++ algItems
  439             , algBuffer = []
  440             , algVars = outs ++ map (applyPatch diff) algVars
  441             }
  442 
  443 addItemToBuffer item = modify'_ $
  444     \alg@AlgBuilder{algBuffer} ->
  445         alg
  446             { algBuffer = item : algBuffer
  447             }
  448 
  449 addItem item vars = modify'_ $
  450     \alg@AlgBuilder{algItems, algVars} ->
  451         alg
  452             { algItems = item : algItems
  453             , algVars = vars ++ algVars
  454             }
  455 
  456 genVar prefix = do
  457     alg@AlgBuilder{algVarGen} <- get
  458     when (null algVarGen) $ error "internal error"
  459     put alg{algVarGen = tail algVarGen}
  460     return $ T.concat [prefix, "_", head algVarGen]
  461 
  462 findAlias var0 = do
  463     AlgBuilder{algItems} <- get
  464     return $ inner var0 algItems
  465     where
  466         inner var [] = var
  467         inner var (Alias{aFrom, aTo} : xs)
  468             | aFrom == var = inner aTo xs
  469         inner var (_ : xs) = inner var xs
  470 
  471 applyPatch diff v =
  472     case find ((== v) . fst) diff of
  473         Just (_, v') -> v'
  474         _ -> v
