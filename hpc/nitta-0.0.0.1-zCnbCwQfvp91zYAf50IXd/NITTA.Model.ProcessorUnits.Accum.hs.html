   49 {- |Type that contains expression:
   50 
   51 @a + b = c@ is exression and it equals:
   52     @[[(False, "a"), (False, "b")], [(False, "c")]]@
   53 
   54 @a + b = c; d - e = f@ is one expression too and it equals:
   55     @[[(False, "a"), (False, "b")], [(False, "c")], [(False, "d"), (True, "d")], [(False, "f")]]@
   56 -}
   57 data Job v x = Job
   58     { -- |Contains future parts expression to eval (c + d = e)
   59       tasks :: [[(Bool, v)]]
   60     , -- |Contain current parts expression (a + b = c)
   61       current :: [[(Bool, v)]]
   62     , -- |Func of this expression
   63       func :: F v x
   64     , -- |Flag indicates when evaluation ended
   65       calcEnd :: Bool
   66     }
   67     deriving (Eq)
   68 
   69 instance (Var v) => Show (Job v x) where
   70     show Job{tasks, current, func, calcEnd} =
   71         [i|Job{tasks=#{ show' tasks }, current=#{ show' current }, func=#{ func }, calcEnd=#{ calcEnd }}|]
   72         where
   73             show' = map (map (second toString))

  107 tryBindJob f@Acc{actions} =
  108     Job
  109         { tasks = concat $ actionGroups actions
  110         , current = []
  111         , func = packF f
  112         , calcEnd = False
  113         }
  114 
  115 actionGroups [] = []
  116 actionGroups as =
  117     let (pushs, as') = span isPush as
  118         (pulls, as'') = span isPull as'
  119      in [ map (\(Push sign (I v)) -> (sign == Minus, v)) pushs
  120         , concatMap (\(Pull (O vs)) -> map (True,) $ elems vs) pulls
  121         ] :
  122         actionGroups as''
  123 
  124 endpointOptionsJob Job{tasks = []} = []
  125 endpointOptionsJob Job{tasks = (t : _), current = []} = map snd t
  126 endpointOptionsJob Job{tasks = (t : ts), current = (c : _)}
  127     | null (t \\ c) && null ts = []
  128     | null $ t \\ c = map snd $ head ts
  129     | otherwise = map snd $ t \\ c
  130 
  131 endpointDecisionJob j@Job{tasks = []} _ = j
  132 endpointDecisionJob j@Job{tasks = tasks@(t : _), current = []} v = j{tasks = updateTasks current' tasks, current = current'}
  133     where
  134         ([(neg, _)], _) = partition ((== v) . snd) t
  135         current' = [[(neg, v)]]
  136 endpointDecisionJob j@Job{tasks = tasks@(t : ts), current = (c : cs)} v
  137     | null $ t \\ c = endpointDecisionJob j{tasks = ts} v
  138     | t \\ c /= t && length t > length c = j{tasks = updateTasks currentInsert tasks, current = currentInsert}
  139     | otherwise = j{tasks = updateTasks currentAdd tasks, current = currentAdd}
  140     where
  141         (lst, _) = partition ((== v) . snd) t
  142         currentInsert = (lst ++ c) : cs
  143         currentAdd = lst : c : cs
  144 
  145 updateTasks (c : _) tasks@(t : ts)
  146     | null $ t \\ c = ts
  147     | otherwise = tasks
  148 updateTasks [] _ = error "Current is null"
  149 updateTasks _ _ = error "Matching error updateTasks"
  150 
  151 toTarget = even . length
  152 toSource = odd . length

  154 instance (VarValTime v x t, Num x) => ProcessorUnit (Accum v x t) v x t where
  155     tryBind f pu@Accum{work}
  156         | Just (Add a b c) <- castF f = Right pu{work = tryBindJob (Acc [Push Plus a, Push Plus b, Pull c]) : work}
  157         | Just (Sub a b c) <- castF f = Right pu{work = tryBindJob (Acc [Push Plus a, Push Minus b, Pull c]) : work}
  158         | Just f'@Acc{} <- castF f = Right pu{work = tryBindJob f' : work}
  159         | otherwise = Left $ "The function is unsupported by Accum: " ++ show f
  160 
  161     process = process_

  163 instance (VarValTime v x t, Num x) => EndpointProblem (Accum v x t) v t where
  164     endpointOptions pu@Accum{currentWork = Just a@Job{tasks, calcEnd}}
  165         | toTarget tasks = targets
  166         | toSource tasks = sources
  167         where
  168             tick = nextTick pu
  169             targets = map (\v -> EndpointSt (Target v) $ TimeConstraint (tick + 1 ... maxBound) (singleton 1)) (endpointOptionsJob a)
  170             sources = [EndpointSt (Source $ fromList (endpointOptionsJob a)) $ TimeConstraint (max tick (tickSource calcEnd) ... maxBound) (1 ... maxBound)]
  171             tickSource True = tick + 1
  172             tickSource _ = tick + 3
  173     endpointOptions p@Accum{work, currentWork = Nothing} =
  174         concatMap (\a -> endpointOptions p{currentWork = Just a}) work
  175     endpointOptions _ = error "Error in matching in endpointOptions function"
  176 
  177     endpointDecision
  178         pu@Accum{currentWork = Just j@Job{tasks}, isInit}
  179         d@EndpointSt{epRole = Target v, epAt}
  180             | not (null tasks) && toTarget tasks =
  181                 let job@Job{tasks = tasks', current = (((neg, _) : _) : _)} = endpointDecisionJob j v
  182                     sel = if isInit then ResetAndLoad neg else Load neg
  183                     (_, process_') = runSchedule pu $ do
  184                         scheduleEndpoint d $ scheduleInstructionUnsafe epAt sel
  185                  in pu
  186                         { process_ = process_'
  187                         , currentWork = Just job{calcEnd = False}
  188                         , isInit = null tasks'
  189                         }
  190     endpointDecision
  191         pu@Accum{currentWork = Just j@Job{tasks, current, func}, process_}
  192         d@EndpointSt{epRole = Source v, epAt}
  193             | not (null current) && toSource tasks =
  194                 let job@Job{tasks = tasks'} = foldl endpointDecisionJob j (elems v)
  195                     a = inf $ stepsInterval $ relatedEndpoints process_ $ variables func
  196                     (_, process_') = runSchedule pu $ do
  197                         endpoints <- scheduleEndpoint d $ scheduleInstructionUnsafe (epAt -1) Out
  198                         when (null tasks') $ do
  199                             high <- scheduleFunction (a ... sup epAt) func
  200                             let low = endpoints ++ map pID (relatedEndpoints process_ $ variables func)
  201                             establishVerticalRelations high low
  202 
  203                         return endpoints
  204                  in pu
  205                         { process_ = process_'
  206                         , currentWork = if null tasks' then Nothing else Just job{calcEnd = True}
  207                         , isInit = null tasks'
  208                         }
  209     endpointDecision pu@Accum{work, currentWork = Nothing} d
  210         | Just job <- getJob work =
  211             endpointDecision pu{work = work \\ [job], currentWork = Just job{calcEnd = False}, isInit = True} d
  212         where
  213             getJob = find (\Job{func} -> d `isIn` func)
  214             e `isIn` f = oneOf (variables e) `member` variables f
  215     endpointDecision pu d = error [i|incorrect decision #{ d } for #{ pretty pu }|]

  261 instance (Var v) => Locks (Accum v x t) v where
  262     locks Accum{currentWork = Nothing} = []
  263     locks Accum{currentWork = Just job, work} = locks' job work
  264         where
  265             locks' Job{tasks = []} _ = []
  266             locks' Job{current = []} _ = []
  267             locks' Job{tasks = (m : ms), current = (r : _)} other =
  268                 [ Lock{lockBy, locked}
  269                 | locked <- concatMap (map snd) ms
  270                 , lockBy <- map snd (m \\ r)
  271                 ]
  272                     ++ [ Lock{lockBy, locked}
  273                        | locked <- concatMap (concatMap (map snd) . tasks) other
  274                        , lockBy <- map snd (m ++ r)
  275                        ]
